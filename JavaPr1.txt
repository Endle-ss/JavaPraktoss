import java.util.Scanner;
import java.text.ParseException;
import java.text.SimpleDateFormat;
import java.util.Date;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.println("Ну здарова");

        System.out.print("Введите вашу фамилию: ");
        String lastName = scanner.nextLine();

        System.out.print("Введите ваше имя: ");
        String firstName = scanner.nextLine();

        System.out.print("Введите ваше отчество: ");
        String middleName = scanner.nextLine();

        System.out.print("Введите вашу дату рождения в формате (dd.MM.yyyy): ");
        String dobString = scanner.nextLine();

        SimpleDateFormat dateFormat = new SimpleDateFormat("dd.MM.yyyy");
        Date dob = null;

        try {
            dob = dateFormat.parse(dobString);
        } catch (ParseException e) {
            System.out.println("Ты лох");
            return;
        }

        int age = calculateAge(dob);

        if (age < 18) {
            System.out.println("Вы несовершеннолетний!");
        } else {
            // Теперь можно использовать age в качестве возраста в годах
            System.out.println("Ваш возраст: " + age + " лет.");

            // Остальная часть кода остается без изменений
            System.out.println("Меню:");
            System.out.println("1. Нахождение P");
            System.out.println("2. Нахождение S");
            System.out.println("3. Нахождение R");
            System.out.println("0. Выход");

            System.out.print("Выберите действие: ");
            int choice = scanner.nextInt();

            switch (choice) {
                case 1:
                    double perimeter = calculate("Периметр", "радиус", 2 * Math.PI);
                    System.out.println("Периметр: " + perimeter);
                    break;
                case 2:
                    double area = calculate("Площадь", "радиус", Math.PI);
                    System.out.println("Площадь: " + area);
                    break;
                case 3:
                    double radius = calculate("Радиус", "периметр", 1 / (2 * Math.PI));
                    System.out.println("Радиус: " + radius);
                    break;
                case 0:
                    System.out.println("Выход");
                    break;
                default:
                    System.out.println("Неправильный выбор. Повторите попытку.");
                    break;
            }
        }
    }

    public static double calculate(String figure, String inputLabel, double multiplier) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Введите " + inputLabel + " " + figure.toLowerCase() + "а: ");
        double value = scanner.nextDouble();
        return value * multiplier;
    }

    public static int calculateAge(Date dob) {
        Date currentDate = new Date();
        long ageInMillis = currentDate.getTime() - dob.getTime();
        return (int) (ageInMillis / 31557600000L); // приблизительное количество миллисекунд в году
    }
}
